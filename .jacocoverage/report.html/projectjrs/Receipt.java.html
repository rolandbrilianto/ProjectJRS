      JOptionPane.showMessageDialog(null, "Terima Kasih");
        int i = JOptionPane.showConfirmDialog(this, "Ingin Memesan Kembali ? ", "Confirm",

        if (i == JOptionPane.YES_OPTION) {
            System_Order f2 = new System_Order();
